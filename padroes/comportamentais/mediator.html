from typing import List, Dict, Any
from datetime import datetime
from enum import Enum

class TipoMensagem(Enum):
    TEXTO = "texto"
    ARQUIVO = "arquivo"
    COMANDO = "comando"
    SISTEMA = "sistema"

class Usuario(ABC):
    def __init__(self, nome: str, departamento: str):
        self.nome = nome
        self.departamento = departamento
        self.mediator = None
    
    def set_mediator(self, mediator: 'ChatMediator'):
        self.mediator = mediator
    
    @abstractmethod
    def receber_mensagem(self, mensagem: Dict[str, Any]):
        pass
    
    @abstractmethod
    def enviar_mensagem(self, conteudo: str, tipo: TipoMensagem = TipoMensagem.TEXTO):
        pass

class ChatMediator(ABC):
    @abstractmethod
    def enviar_mensagem(self, remetente: Usuario, conteudo: str, tipo: TipoMensagem):
        pass
    
    @abstractmethod
    def adicionar_usuario(self, usuario: Usuario):
        pass
    
    @abstractmethod
    def criar_grupo(self, nome: str, usuarios: List[Usuario]):
        pass

class SistemaChatEmpresarial(ChatMediator):
    def __init__(self, nome_empresa: str):
        self.nome_empresa = nome_empresa
        self.usuarios: Dict[str, Usuario] = {}
        self.grupos: Dict[str, List[Usuario]] = {}
        self.historico: List[Dict[str, Any]] = []
        self.online_users = set()
    
    def adicionar_usuario(self, usuario: Usuario):
        self.usuarios[usuario.nome] = usuario
        usuario.set_mediator(self)
        print(f"‚úÖ {usuario.nome} ({usuario.departamento}) adicionado ao chat")
    
    def criar_grupo(self, nome: str, usuarios: List[Usuario]):
        self.grupos[nome] = usuarios
        print(f"üë• Grupo '{nome}' criado com {len(usuarios)} usu√°rios")
    
    def enviar_mensagem(self, remetente: Usuario, conteudo: str, tipo: TipoMensagem = TipoMensagem.TEXTO):
        mensagem = {
            'remetente': remetente.nome,
            'conteudo': conteudo,
            'tipo': tipo,
            'timestamp': datetime.now(),
            'departamento': remetente.departamento
        }
        
        self.historico.append(mensagem)
        
        print(f"\nüì® MEDIATOR: Mensagem de {remetente.nome}")
        print(f"   üìù Conte√∫do: {conteudo}")
        print(f"   üè∑Ô∏è  Tipo: {tipo.value}")
        
        # L√≥gica de roteamento baseada no tipo de mensagem e departamento
        if tipo == TipoMensagem.COMANDO:
            self._processar_comando(remetente, conteudo)
        elif tipo == TipoMensagem.SISTEMA:
            self._broadcast_sistema(mensagem)
        else:
            self._rotear_mensagem(remetente, mensagem)
    
    def _processar_comando(self, remetente: Usuario, comando: str):
        comando = comando.lower().strip()
        
        if comando == "/online":
            online = ", ".join(self.online_users)
            resposta = f"üë§ Usu√°rios online: {online if online else 'Nenhum'}"
            remetente.receber_mensagem({
                'remetente': 'Sistema',
                'conteudo': resposta,
                'tipo': TipoMensagem.SISTEMA,
                'timestamp': datetime.now()
            })
        
        elif comando == "/historico":
            total = len(self.historico)
            resposta = f"üìä Total de mensagens: {total}"
            remetente.receber_mensagem({
                'remetente': 'Sistema',
                'conteudo': resposta,
                'tipo': TipoMensagem.SISTEMA,
                'timestamp': datetime.now()
            })
        
        elif comando.startswith("/grupo "):
            grupo_nome = comando[7:]
            if grupo_nome in self.grupos:
                membros = [user.nome for user in self.grupos[grupo_nome]]
                resposta = f"üë• Membros do grupo '{grupo_nome}': {', '.join(membros)}"
                remetente.receber_mensagem({
                    'remetente': 'Sistema',
                    'conteudo': resposta,
                    'tipo': TipoMensagem.SISTEMA,
                    'timestamp': datetime.now()
                })
        
        else:
            remetente.receber_mensagem({
                'remetente': 'Sistema',
                'conteudo': f"‚ùå Comando desconhecido: {comando}",
                'tipo': TipoMensagem.SISTEMA,
                'timestamp': datetime.now()
            })
    
    def _broadcast_sistema(self, mensagem: Dict[str, Any]):
        print("   üì¢ Broadcast para todos os usu√°rios")
        for usuario in self.usuarios.values():
            usuario.receber_mensagem(mensagem)
    
    def _rotear_mensagem(self, remetente: Usuario, mensagem: Dict[str, Any]):
        # Roteamento inteligente baseado em departamento e conte√∫do
        conteudo = mensagem['conteudo'].lower()
        
        if any(palavra in conteudo for palavra in ['bug', 'erro', 'problema']):
            self._enviar_para_ti(mensagem)
        elif any(palavra in conteudo for palavra in ['venda', 'cliente', 'contrato']):
            self._enviar_para_vendas(mensagem)
        elif any(palavra in conteudo for palavra in ['design', 'interface', 'ux']):
            self._enviar_para_design(mensagem)
        else:
            self._enviar_para_departamento(remetente.departamento, mensagem)
    
    def _enviar_para_ti(self, mensagem: Dict[str, Any]):
        print("   üñ•Ô∏è  Roteando para departamento de TI")
        for usuario in self.usuarios.values():

                usuario.receber_mensagem(mensagem)
    
    def _enviar_para_vendas(self, mensagem: Dict[str, Any]):

        for usuario in self.usuarios.values():
            if usuario.departamento == "Vendas":
                usuario.receber_mensagem(mensagem)
    
    def _enviar_para_design(self, mensagem: Dict[str, Any]):
        print("   üé® Roteando para departamento de Design")
        for usuario in self.usuarios.values():
            if usuario.departamento == "Design":
                usuario.receber_mensagem(mensagem)
    
    def _enviar_para_departamento(self, departamento: str, mensagem: Dict[str, Any]):
        print(f"   üè¢ Roteando para departamento {departamento}")
        for usuario in self.usuarios.values():
            if usuario.departamento == departamento:
                usuario.receber_mensagem(mensagem)
    
    def conectar_usuario(self, usuario: Usuario):
        self.online_users.add(usuario.nome)
