class ComandoCena(Command):
    def __init__(self, nome: str, comandos: List[Command]):
        self.nome = nome
        self.comandos = comandos
        self.comandos_executados = []
    
    def execute(self) -> None:
        print(f"üé≠ EXECUTANDO CENA: {self.nome}")
        self.comandos_executados = self.comandos.copy()
        for comando in self.comandos:
            comando.execute()
    
    def undo(self) -> None:
        print(f"‚Ü©Ô∏è  DESFAZENDO CENA: {self.nome}")
        for comando in reversed(self.comandos_executados):
            comando.undo()
    
    def get_description(self) -> str:
        return f"Cena: {self.nome} ({len(self.comandos)} comandos)"

class ControladorCasa:

        while self.comandos_pendentes:
            comando = self.comandos_pendentes.pop(0)
            comando.execute()
            self.historico.append(comando)
            print("")
        
        print("‚úÖ Todos os comandos executados!")
    
    def executar_comando(self, comando: Command) -> None:
        print(f"\nüéØ EXECUTANDO COMANDO: {comando.get_description()}")
