class ContextoConsulta:
    def __init__(self, dados_musicais: Dict[str, Any]):
        self.dados_musicais = dados_musicais
        self.resultado = None
        self.variaveis = {}

        return self.variaveis.get(nome)

# Abstract Expression
class Expressao(ABC):

    @abstractmethod
    def __str__(self) -> str:

# Terminal Expressions

class ExpressaoArtista(Expressao):
    def __init__(self, nome_artista: str):
        self.nome_artista = nome_artista
    
    def interpretar(self, contexto: ContextoConsulta) -> Any:
        dados = contexto.get_dados()

    def __str__(self) -> str:
        return f"Artista('{self.nome_artista}')"

class ExpressaoAlbum(Expressao):

class ExpressaoFiltroAno(Expressao):
    def __init__(self, expressao: Expressao, operador: str, ano: Expressao):
        self.expressao = expressao
        self.operador = operador  # '>', '<', '=='
        self.ano = ano
    
    def interpretar(self, contexto: ContextoConsulta) -> Any:
        resultado_base = self.expressao.interpretar(contexto)
        ano_alvo = self.ano.interpretar(contexto)
        
        if not resultado_base or not ano_alvo:
            return None
        
        # Filtra baseado no ano
        if isinstance(resultado_base, list):
            if self.operador == '>':
                return [item for item in resultado_base if item.get('ano', 0) > ano_alvo]
            elif self.operador == '<':
                return [item for item in resultado_base if item.get('ano', 0) < ano_alvo]
            elif self.operador == '==':
                return [item for item in resultado_base if item.get('ano', 0) == ano_alvo]
        elif isinstance(resultado_base, dict):
            ano_item = resultado_base.get('ano')
            if ano_item:
                if self.operador == '>' and ano_item > ano_alvo:
                    return resultado_base
                elif self.operador == '<' and ano_item < ano_alvo:
                    return resultado_base
                elif self.operador == '==' and ano_item == ano_alvo:
                    return resultado_base
        return None
    
    def __str__(self) -> str:
        return f"FiltroAno({self.expressao} {self.operador} {self.ano})"

class ExpressaoEstatistica(Expressao):

        resultado = self.expressao.interpretar(contexto)
        
        if not resultado:
            return None
        
        if self.estatistica == 'count':
            if isinstance(resultado, list):
                return len(resultado)
            else:
                return 1 if resultado else 0
        
        elif self.estatistica == 'sum':
            if isinstance(resultado, list):
                # Soma streams ou vendas
                return sum(item.get('streams', 0) for item in resultado)
            elif isinstance(resultado, dict):
                return resultado.get('streams', 0)
        
        elif self.estatistica == 'avg':
            if isinstance(resultado, list) and resultado:
                streams = [item.get('streams', 0) for item in resultado]
                return sum(streams) / len(streams)
            elif isinstance(resultado, dict):
                return resultado.get('streams', 0)
        
        elif self.estatistica == 'max':
            if isinstance(resultado, list) and resultado:
                return max(item.get('streams', 0) for item in resultado)
            elif isinstance(resultado, dict):
                return resultado.get('streams', 0)
        
        elif self.estatistica == 'min':
            if isinstance(resultado, list) and resultado:
                return min(item.get('streams', 0) for item in resultado)
            elif isinstance(resultado, dict):
                return resultado.get('streams', 0)
        
        return None

        return f"Estatistica({self.estatistica}, {self.expressao})"

class ExpressaoComposta(Expressao):

        resultados = []
        for expressao in self.expressoes:

            if resultado:
                if self.operador == 'AND':
                    resultados.append(resultado)
                elif self.operador == 'OR':
                    return resultado  # Retorna no primeiro verdadeiro

                return None  # AND falha se alguma for falsa

        return None

        expressoes_str = " & ".join(str(expr) for expr in self.expressoes)
        return f"Composta({expressoes_str})"
