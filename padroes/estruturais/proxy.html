class CacheProxy(DatabaseOperations):
    def __init__(self, real_database: DatabaseOperations):
        self._real_database = real_database
        self._query_cache: Dict[str, tuple] = {}  # sql_hash -> (result, timestamp)
        self._user_cache: Dict[int, tuple] = {}   # user_id -> (data, timestamp)
        self._cache_duration = 300  # 5 minutos em segundos
    
    def _is_cache_valid(self, timestamp: float) -> bool:
        return (datetime.now().timestamp() - timestamp) < self._cache_duration
    
    def _get_cache_key(self, sql: str, params: tuple = None) -> str:
        key_data = sql + str(params)
        return hashlib.md5(key_data.encode()).hexdigest()
    
    def query(self, sql: str, params: tuple = None) -> List[Dict[str, Any]]:
        cache_key = self._get_cache_key(sql, params)
        
        if cache_key in self._query_cache:
            result, timestamp = self._query_cache[cache_key]
            if self._is_cache_valid(timestamp):
                print("üíæ CACHE: Retornando resultado do cache")
                return result
        
        print("üíæ CACHE: Cache miss, executando query real")
        result = self._real_database.query(sql, params)
        self._query_cache[cache_key] = (result, datetime.now().timestamp())
        return result
    
    def execute(self, sql: str, params: tuple = None) -> int:
        # Opera√ß√µes de escrita invalidam cache
        if "UPDATE" in sql.upper() or "INSERT" in sql.upper() or "DELETE" in sql.upper():
            print("üíæ CACHE: Invalidando cache devido a opera√ß√£o de escrita")
            self._query_cache.clear()
            self._user_cache.clear()
        
        return self._real_database.execute(sql, params)
    
    def get_user_data(self, user_id: int) -> Dict[str, Any]:
        if user_id in self._user_cache:
            data, timestamp = self._user_cache[user_id]
            if self._is_cache_valid(timestamp):
                print("üíæ CACHE: Retornando dados do usu√°rio do cache")
                return data
        
        print("üíæ CACHE: Cache miss, buscando dados do usu√°rio")
        data = self._real_database.get_user_data(user_id)
        self._user_cache[user_id] = (data, datetime.now().timestamp())
        return data
    
    def update_user_data(self, user_id: int, data: Dict[str, Any]) -> bool:
        # Atualiza√ß√£o invalida cache do usu√°rio
        if user_id in self._user_cache:
            del self._user_cache[user_id]
        
        result = self._real_database.update_user_data(user_id, data)
        
        if result:
            # Atualiza cache com novos dados
            self._user_cache[user_id] = (data, datetime.now().timestamp())
        
        return result

class LoggingProxy(DatabaseOperations):
    def __init__(self, real_database: DatabaseOperations, log_file: str = "database_operations.log"):
        self._real_database = real_database
        self._log_file = log_file
        self._log_operations = []
    
    def _log_operation(self, operation: str, details: str, success: bool = True):
        timestamp = datetime.now().isoformat()
        log_entry = {
            "timestamp": timestamp,
            "operation": operation,
            "details": details,
            "success": success
        }
        self._log_operations.append(log_entry)
        
        print(f"üìù LOG: {operation} - {details} - {'SUCESSO' if success else 'FALHA'}")
        
        # Tamb√©m escreve em arquivo (em produ√ß√£o)
        try:
            with open(self._log_file, 'a') as f:
                f.write(json.dumps(log_entry) + '\n')
        except:
            pass  # Em caso de erro de arquivo, apenas ignora
    
    def query(self, sql: str, params: tuple = None) -> List[Dict[str, Any]]:

        try:

    def execute(self, sql: str, params: tuple = None) -> int:
        self._log_operation("EXECUTE", f"SQL: {sql}, Params: {params}")
        
        try:
            result = self._real_database.execute(sql, params)
            self._log_operation("EXECUTE", f"Linhas afetadas: {result}")
            return result
        except Exception as e:
            self._log_operation("EXECUTE", f"Erro: {str(e)}", False)
            raise
    
    def get_user_data(self, user_id: int) -> Dict[str, Any]:
        self._log_operation("GET_USER_DATA", f"User ID: {user_id}")
        
        try:
            result = self._real_database.get_user_data(user_id)
            self._log_operation("GET_USER_DATA", f"Dados recuperados: {bool(result)}")
            return result
        except Exception as e:
            self._log_operation("GET_USER_DATA", f"Erro: {str(e)}", False)
            raise
    
    def update_user_data(self, user_id: int, data: Dict[str, Any]) -> bool:
        self._log_operation("UPDATE_USER_DATA", f"User ID: {user_id}, Data: {data}")
        
        try:
            result = self._real_database.update_user_data(user_id, data)
            self._log_operation("UPDATE_USER_DATA", f"Sucesso: {result}")
            return result
        except Exception as e:
            self._log_operation("UPDATE_USER_DATA", f"Erro: {str(e)}", False)
            raise
    
    def get_audit_log(self) -> List[Dict[str, Any]]:
        return self._log_operations.copy()

            self._real_database = RealDatabase()

        self._initialize()
        return self._real_database.query(sql, params)
    
    def execute(self, sql: str, params: tuple = None) -> int:
        self._initialize()
        return self._real_database.execute(sql, params)
    
    def get_user_data(self, user_id: int) -> Dict[str, Any]:
        self._initialize()
        return self._real_database.get_user_data(user_id)
    
    def update_user_data(self, user_id: int, data: Dict[str, Any]) -> bool:
        self._initialize()
        return self._real_database.update_user_data(user_id, data)
    
    def close(self):
        if self._initialized and self._real_database:
